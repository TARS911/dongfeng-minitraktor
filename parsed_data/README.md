### Таблица: `dongfeng_parts`

```sql
CREATE TABLE dongfeng_parts (
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_name VARCHAR(500) NOT NULL,
    price_rubles DECIMAL(10, 2),
    sku_code VARCHAR(50),
    in_stock ENUM('yes', 'no', 'unknown'),
    category VARCHAR(100),
    subcategory VARCHAR(100),
    product_url TEXT,
    date_added TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

);
```

```sql

-- По категориям
SELECT category, COUNT(*) as count 
FROM dongfeng_parts 
GROUP BY category

-- По наличию

FROM dongfeng_parts 
GROUP BY in_stock;

-- Статистика по ценам

    MAX(price_rubles) as max_price,
    MIN(price_rubles) as min_price,
```
